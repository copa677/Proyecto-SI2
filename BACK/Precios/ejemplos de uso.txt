üêç EJEMPLOS EN PYTHON CON REQUESTS

import requests

BASE_URL = "http://localhost:8000/api/precios"

# Listar precios activos
response = requests.get(f"{BASE_URL}/precios/")
print("Precios activos:", response.json())

# Crear nuevo precio
nuevo_precio = {
    "decripcion": "Polera t√©cnica",
    "material": "Dry-fit",
    "talla": "M",
    "precio_base": "38.99"
}
response = requests.post(f"{BASE_URL}/precios/create/", json=nuevo_precio)
print("Precio creado:", response.json())

# Buscar precios
response = requests.get(f"{BASE_URL}/precios/search/?q=algod√≥n")
print("B√∫squeda:", response.json())

# Actualizar precio
precio_actualizado = {
    "decripcion": "Polera t√©cnica premium",
    "material": "Dry-fit mejorado",
    "talla": "M",
    "precio_base": "42.50",
    "activo": True
}
response = requests.put(f"{BASE_URL}/precios/update/1/", json=precio_actualizado)
print("Precio actualizado:", response.json())

# Eliminar precio
response = requests.delete(f"{BASE_URL}/precios/delete/2/")
print("Eliminaci√≥n:", response.json())
